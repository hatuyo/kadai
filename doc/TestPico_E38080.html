def test_ummm
  @pc.I_have_a_right("Pen")
  assert_equal("Pen", @pc.right)
  @pc.I_have_a_left("Painappow")
  assert_equal("Painappow", @pc.left)
  @pc.ummm(@pc.right,@pc.left)
  assert_equal("PenPainappow", @pc.result1)
end
